BOJ 11050 이항계수 1

접근하기

이항계수의 성질 중 점화식 이용

4C3은? 
1고른 상태에서 남은 3개중 2개 고르기 = 3C2
1 2 3 4     

1안고른상태에서 남은 3개중  3개 고르기 = 3C3
1 2 3 4

dp[i][j] = dp[i-1][j-1] + dp[i-1][j]

코드

#include <bits/stdc++.h>
using namespace std;
int N, K;

int dp[12][12];

int main(int argc, char* argv[])
{
    scanf("%d %d", &N, &K);
    dp[0][0] = 1;

    for(int i=0;i<=N;i++) {
        dp[i][0] = 1;
    }

    for(int i=1;i<=N;i++) {
        for(int j=1;j<=min(i, K);j++) {
            dp[i][j] = dp[i-1][j-1] +  dp[i-1][j];
        }
    }

    printf("%d\n", dp[N][K]);
    return 0;
}
