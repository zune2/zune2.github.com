BOJ 21758 꿀 따기

접근하기

1. 벌통이 맨 오른쪽에 있는 경우

2. 벌통이 맨 왼쪽에 있는 경우

3. 벌통이 가운데 어딘가 있는 경우

코드

#include <bits/stdc++.h>
using namespace std;

int N;
int a[100010];
int s[100010];

int main(int argc, char* argv[])
{
    freopen("input.txt", "r", stdin);
    scanf("%d", &N);
    for(int i=1;i<=N;i++) {
        scanf("%d", &a[i]);
    }

    s[1] = a[1];
    for(int i=2;i<=N;i++) {
        s[i] = s[i-1] + a[i];
    }
    int ans = 0;

    for(int i=2;i<N;i++) {
        ans = max(ans, s[N] - a[1] - a[i] + s[N] - s[i-1] - a[i]);
        ans = max(ans, s[N] - a[N] - a[i] + s[i-1]);
        ans = max(ans, s[i] - a[1] + s[N] - s[i-1] - a[N]);
    }

    printf("%d\n", ans);
    return 0;
}
