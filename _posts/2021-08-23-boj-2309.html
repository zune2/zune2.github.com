int main(int argc, char *argv[])
{
    for(int i=0;i<9;i++) {
        scanf("%d", &a[i]);    
    }
    sort(a, a+9);

    for(unsigned int i=0; i<(1<<9);i++)  {
        int total = 0;
        int cnt = 0;
        for(unsigned int j=0;j<9;j++) {
            if( (i & (1<<j))){
                total += a[j]; 
                cnt += 1;
            }
        }
        if (total == 100 && cnt == 7) {
            for(unsigned int j=0;j<9;j++) {
                if( (i & (1<<j))){
                    printf("%d\n", a[j]);
                }
            }
            break;
        }
    }
    return 0;
}
