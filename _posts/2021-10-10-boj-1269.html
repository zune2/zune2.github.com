BOJ 1269 대칭 차집합

접근하기

  • set1, set2에 각각 데이터를 저장
  • set1에서 B를 빼고 카운트
  • set2에서 A를 빼고 카운트

코드

#include <cstdio>
#include <set>
#include <algorithm>

using namespace std;

int a[200020];
int b[200020];

int N, M;
int main(int argc, char* argv[])
{
    scanf("%d %d", &N, &M);
    set<int> sa;
    for(int i=0;i<N;i++) {
        scanf("%d", &a[i]);
        sa.insert(a[i]);
    }

    set<int> sb;
    for(int i=0;i<M;i++) {
        scanf("%d", &b[i]);
        sb.insert(b[i]);
    }

    vector<int> v;
    for(int i=0;i<N;i++) {
        if (sb.find(a[i]) != sb.end()) {
            sa.erase(a[i]);
            v.push_back(a[i]);
        }
    }
    int reta = sa.size();

    for(auto &o :v) {
        sa.insert(o);
    }

    for(int i=0;i<M;i++) {
        if (sa.find(b[i]) != sa.end()) {
            sb.erase(b[i]);
        }
    }

    printf("%d\n", reta + (int)sb.size());
    return 0;
}
