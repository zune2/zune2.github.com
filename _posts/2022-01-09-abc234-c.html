C - Happy New Year!

접근하기

3 -> 11 로 생각가능

11 -> 1011 로 생각 가능

2진수로 변환하여, 1을 2로 출력, 0을 0으로 string을 만들어 뒤집기

코드

#include <bits/stdc++.h>
using namespace std;

long long int N;

int main(int argc, char* argv[])
{
    string ans;
    scanf("%lld", &N);

    while(N) {
        int nam = N %2; 
        if (nam == 0) {
            ans.push_back('0');
        } else {
            ans.push_back('2');
        }
        N = N / 2;
    }
    reverse(ans.begin(), ans.end());
    cout << ans << '\n';
    return 0;
}
