BOJ 1152 단어의 개수

접근하기

  • getline으로 줄 단위로 받아서 string에 저장
  • istringstream을 이용해서, token으로 분리
  • vector에 token을 저장하지 않고, cnt를 세면 된다.

코드

#include <iostream>
#include <sstream>
#include <string>

using namespace std;


int main(int argc, char* argv[])
{
    string s;
    getline(cin >> ws, s);

    char split_char = ' ';
    istringstream split(s);
    //vector<string> tok;
    int cnt = 0;
    for(string each; getline(split, each, split_char);cnt+=1);
    cout << cnt << endl;
    return 0;
}
