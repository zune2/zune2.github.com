BOJ 2965 캥거루 세마리

  • 정수 좌표에 3마리의 캥거루가 다른 점에 있을때, 바깥쪽 캥거루가 이동 가능할 때 최대한 이동하는 수 구하기

접근하기

  • 차이를 세고 -1 하면 된다는 성질 발견

코드

#include <cstdio>
#include <algorithm>
using namespace std;

int a, b, c;

int main(int argc, char* argv[])
{
    scanf("%d %d %d", &a, &b, &c);
    int val = max(abs(a-b), abs(b-c));
    printf("%d\n", val-1);
    return 0;
}
