** BOJ 14425 문자열 집합

** 접근하기
- 집합 S에 같은 문자열이 여러 번 주어지는 경우는 없으므로, STL set을 에 넣었음.
- cnt를 이용해서 count함

** 코드
#+BEGIN_SRC cpp
#include <cstdio>
#include <string>
#include <set>
using namespace std;

int N, M;
set<string> mp;

int main(int argc, char* argv[])
{
    scanf("%d %d", &N, &M);

    char str[510];
    for(int i=0;i<N;i++) {
        scanf("%s", str);
        mp.insert(string(str));
    }
    
    int cnt = 0;
    for(int i=0;i<M;i++) {
        scanf("%s", str);
        if (mp.find(string(str)) != mp.end()) cnt++;
    }
    printf("%d\n", cnt);
    return 0;
}
#+END_SRC
