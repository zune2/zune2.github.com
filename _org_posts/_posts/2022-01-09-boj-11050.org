dp[i][j] = dp[i-1][j-1] + dp[i-1][j]

#+END_EXAMPLE

** 코드
#+BEGIN_SRC cpp
#include <bits/stdc++.h>
using namespace std;
int N, K;

int dp[12][12];

int main(int argc, char* argv[])
{
    scanf("%d %d", &N, &K);
    dp[0][0] = 1;

    for(int i=0;i<=N;i++) {
        dp[i][0] = 1;
    }

    for(int i=1;i<=N;i++) {
        for(int j=1;j<=min(i, K);j++) {
            dp[i][j] = dp[i-1][j-1] +  dp[i-1][j];
        }
    }

    printf("%d\n", dp[N][K]);
    return 0;
}
#+END_SRC
