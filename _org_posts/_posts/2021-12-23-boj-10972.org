** BOJ 10972 다음 순열
- 다음으로 나오는 순열 구하기

** 접근하기
- STL next_permutation을 이용

** 코드 - STL next permutation 이용함
#+BEGIN_SRC cpp
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;

int N;

int main(int argc, char* argv[])
{
    scanf("%d", &N);
    vector<int> a(N);
    int num;
    for(int i=0;i<N;i++) {
        scanf("%d", &num);
        a[i] = num;
    }
    if (next_permutation(a.begin(), a.end())) {
        for(auto& o:a) {
            printf("%d ", o);
        } 
        printf("\n");
    } else {
        printf("-1\n");
    }
    return 0;
}
#+END_SRC
