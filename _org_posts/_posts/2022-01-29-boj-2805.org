** 코드
#+BEGIN_SRC cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int N, M;
int a[1000010];

bool check(int mid)
{
    ll sum = 0;
    for(int i=0;i<N;i++) {
        int val = a[i] - mid;
        if (val > 0) {
            sum += val;    
        }
    } 
    return (sum >= M);
}

int main(int argc, char* argv[])
{
    scanf("%d %d", &N, &M);
    
    for(int i=0;i<N;i++) {
        scanf("%d", &a[i]);
    }

    int s = 0;
    int e = 1000000010;
    int ans = 0;
    while(s+1 < e) {
        int mid = (s+e)/2;

        if (check(mid)) {
            ans = mid;
            s = mid; 
        } else {
            e = mid; 
        }
    }
    printf("%d\n", ans);
    return 0;
}
#+END_SRC
