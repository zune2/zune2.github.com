** 코드 - sqrt 이용
#+BEGIN_SRC cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main(int argc, char* argv[])
{
    ll val;
    scanf("%lld", &val); 
    ll ret = sqrt(val);
    if (ret * ret == val)
        printf("%lld\n", ret);
    else
        printf("%lld\n", ret + 1);
    return 0;
}
#+END_SRC

** 코드 - 이분탐색 이용
#+BEGIN_SRC cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll N;

int main(int argc, char* argv[])
{
    scanf("%lld", &N); 

    ll s = -1;
    ll e = sqrt(numeric_limits<long long>::max()) + 1;

    while(s+1 < e) {
        ll mid = (s + e) / 2;
        if ( mid*mid >= N) {
            e = mid;   // 최대값의 최소
        } else {
            s = mid;
        }
    } 
    printf("%lld\n", e);
    return 0;
}
#+END_SRC
