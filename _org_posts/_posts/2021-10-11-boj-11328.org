** BOJ 11328 Strfry 

** 접근하기
- STL string을 sort로 정렬할 수 있음.
- 두 string을 sort한 후, 비교
  
** 코드
#+BEGIN_SRC cpp
#include <iostream>
//#include <cstdio>
#include <string>
#include <algorithm>
using namespace std;

int N;

int main(int argc, char* argv[])
{
    freopen("input.txt", "r", stdin);
    cin >> N;
    string s1, s2;
    for(int i=0;i<N;i++) {
        cin >> s1 >> s2; 

        sort(s1.begin(), s1.end());
        sort(s2.begin(), s2.end());
        if (s1 == s2)
            cout << "Possible" << endl;
        else
            cout << "Impossible" << endl;
    }

    return 0;
}
#+END_SRC
