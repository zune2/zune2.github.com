** BOJ 9536 여우는 어떻게 울지?

** 접근하기
- 점화식이 주어질 째, top-down으로 구성하기
- 수학 기호의 이해
- STL map을 이용해서 dp 배열을 만들 수 있음

** 코드
#+BEGIN_SRC cpp
#include <cstdio>
#include <map>
using namespace std;

map <long long, long long> dp;

long long N;
int P, Q;

long long f(long long i)
{
    if (i == 0) return 1;
    if (dp[i]) return dp[i];

    dp[i] = f(i/P) + f(i/Q);
    return dp[i];
}

int main(int argc, char* argv[])
{
    scanf("%lld %d %d", &N, &P, &Q);

    long long ret = f(N);
    printf("%lld\n", ret); 
    return 0;
}
#+END_SRC
