** STL set 
- 데이터의 중복없이, 순서대로 정렬되어 저장하는 자료구조
- Red Black tree 기반

** 예제

#+BEGIN_SRC cpp
#include <iostream>
#include <algorithm>
#include <set>
using namespace std;

int main(int argc, char* argv[])
{
    set <int> s;

    //for(int i=1;i<=10;i++)
    s.insert(1);
    s.insert(1);
    s.insert(2);
    s.insert(2);
    s.insert(3);
    s.insert(3);

    for(set<int>::iterator it=s.begin();it!=s.end();it++) {
        cout << *it << endl;
    }
    return 0;
}
