** 코드
#+BEGIN_SRC cpp
#include <bits/stdc++.h>
using namespace std;

int T, N;

void solve()
{
    scanf("%d", &N);
    vector<pair<int, int>> v(N);
    for(int i=0;i<N;i++) {
        scanf("%d %d", &v[i].first, &v[i].second);
    }
    sort(v.begin(), v.end());

    int cnt = 1;
    int mv = v[0].second;
    for(int i=1;i<(int)v.size();i++) {
        if (mv > v[i].second) {
            mv = v[i].second;
            cnt += 1;
        }
    }
    printf("%d\n", cnt);
}

int main(int argc, char* argv[])
{
    scanf("%d", &T);
    for(int tc=1;tc<=T;tc++) {
        solve();
    }
    return 0;
}
#+END_SRC
