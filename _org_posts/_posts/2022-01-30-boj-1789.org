** 코드 - 그리디
#+BEGIN_SRC cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll N;

int main(int argc, char* argv[])
{
    scanf("%lld", &N);

    long long sum = 0;
    long long i = 1;
    while(1) {
        sum += i;
        if (sum > N) {
            printf("%lld\n", i - 1);
            return 0 ;
        }
        i += 1;
    }
    return 0;
}
