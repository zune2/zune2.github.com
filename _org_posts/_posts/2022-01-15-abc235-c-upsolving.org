** C - The Kth Time Query

** 접근하기
#+BEGIN_EXAMPLE
- map 배열을 만들어서 x값별로 추가
- kth가 map[x].size보다 작으면, 출력, 없으면 -1 
#+END_EXAMPLE

** 코드
#+BEGIN_SRC cpp
#include <bits/stdc++.h>
using namespace std;
int N, Q;

int a[200020];
int x, kth;

map<int, vector<int>> mp;

int main(int argc, char* argv[])
{
    scanf("%d %d", &N, &Q);
    for(int i=1;i<=N;i++) {
        scanf("%d", &a[i]);
        mp[a[i]].push_back(i);
    }

    for(int i=0;i<Q;i++) {
        scanf("%d %d", &x, &kth);
        
        int size = mp[x].size();
        if (kth <= size) {
            printf("%d\n", mp[x][kth-1]);
        } else {
            printf("-1\n");
        }
    }
    return 0;
}
#+END_SRC
