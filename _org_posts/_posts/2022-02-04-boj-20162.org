** BOJ 20162 간식 파티
- dp로 LIS풀기

** 접근하기
#+BEGIN_EXAMPLE
서울이는 전에 먹었던 간식보다 더 맛있는 간식만 먹는다

5
4 5 1 2 3

4 먹고 5를 먹어서 9가 만족의 최대

누적값이 최대가 되는 LIS(N) 문제
#+END_EXAMPLE
** 코드
#+BEGIN_SRC cpp
#include <bits/stdc++.h>
using namespace std;
#define FASTIO ios_base::sync_with_stdio(false); cin.tie(NULL);
typedef long long ll;
#define INF 1e9

#define ARRSIZE 1010

ll N;
ll a[ARRSIZE];
ll dp[ARRSIZE];

int main(int argc, char* argv[])
{
    FASTIO
    cin >> N;
    for(int i=1;i<=N;i++) {
        cin >> a[i];
    }

    for(int i=1;i<=N;i++) {
        dp[i] = a[i];
    }

    for(int i=1;i<=N;i++) {
        for(int j=1;j<=i;j++) {
            if (a[i] > a[j] && dp[i] < dp[j] + a[i]) {
                dp[i] = dp[j] + a[i];
            }
        }
    }
    ll ans = dp[1];
    for(int i=1;i<=N;i++) {
        ans = max(ans, dp[i]);
    }

    cout << ans << '\n';
    return 0;
}
#+END_SRC
