** BOJ 11931 수 정렬하기 4

** 접근하기
- STL sort를 이용하기. 역순을 위해 cmp함수를 추가해주기

** 코드
#+BEGIN_SRC cpp
#include <cstdio>
#include <algorithm>

using namespace std;

int N;
int a[1000010];

bool cmp(int& p1, int& p2)
{
    return p1 > p2;
}

int main(int argc, char* argv[])
{
    scanf("%d", &N);  
    for(int i=0;i<N;i++)
        scanf("%d", &a[i]);

    sort(a, a+N, cmp);
    for(int i=0;i<N;i++)
        printf("%d\n", a[i]);
    return 0;
}
#+END_SRC
