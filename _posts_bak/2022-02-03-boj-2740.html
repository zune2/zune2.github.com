BOJ 2740 행렬 곱셈

접근하기

  • 행렬 곱셈 수학식을 2차원 배열로 코딩
A = 3 x 2 

B = 2 x 3

C = 3 x 3 행렬이 된다.

y-> A세로 = 3
x-> B가로 = 3

코드

#include <bits/stdc++.h>
using namespace std;
#define FASTIO ios_base::sync_with_stdio(false); cin.tie(NULL);
typedef long long ll;
#define INF 1e9

int N, M, K;
int a[110][110];
int b[110][110];
int c[110][110];

int main(int argc, char* argv[])
{
    FASTIO
    cin >> N >> M;
    for(int i=1;i<=N;i++) {
        for(int j=1;j<=M;j++) {
            cin >> a[i][j];
        }
    }

    cin >> M >> K;
    for(int i=1;i<=M;i++) {
        for(int j=1;j<=K;j++) {
            cin >> b[i][j];
        }
    }

    for(int i=1;i<=N;i++) {          // 1 ~ 3
        for(int j=1;j<=K;j++) {      // 1 ~ 3
            for(int k=1;k<=M;k++) {  // 1 ~ 2
                c[i][j] += a[i][k]*b[k][j];
            }
        }
    }

    for(int i=1;i<=N;i++) {
        for(int j=1;j<=K;j++) {
            cout << c[i][j] << " ";
        }
        cout << '\n';
    }

    return 0;
}
