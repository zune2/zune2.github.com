BOJ 11722 가장 긴 감소하는 부분 수열

접근하기

가장 긴 증가하는 부분수열 반대로 하기

코드

#include <bits/stdc++.h>
using namespace std;
#define FASTIO ios_base::sync_with_stdio(false); cin.tie(NULL);
typedef long long ll;
#define INF 1e9

int N;
int a[1010];
int dp[1010];

int main(int argc, char* argv[])
{
    FASTIO
#ifndef ONLINE_JUDGE
    freopen("input.txt", "r", stdin);
#endif
    cin >> N;
    for(int i=1;i<=N;i++) {
        cin >> a[i];
    }

    int ans = 1;
    for(int i=1;i<=N;i++) {
        dp[i] = 1;
        for(int j=1;j<=i;j++) {
            if (a[i] < a[j] && dp[i] < dp[j] + 1) {
                dp[i] = dp[j] + 1;
                ans = max(ans, dp[i]);
            } 
        }
    }
    cout << ans << '\n';
    return 0;
}
